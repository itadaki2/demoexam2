### 3. `ProductPartial.cs` (Создай в папке `Entities`)

Это магия, которая высчитает итоговую цену со скидкой и раскрасит фон строк, если товара нет (серый) или скидка > 25% (#23E1EF).

```csharp
namespace ChitayApp.Entities;

public partial class Product
{
    // Итоговая цена
    public decimal FinalPrice => Discount > 0 ? Price - (Price * (Discount ?? 0) / 100) : Price;
    
    // Показывать ли перечеркнутую старую цену
    public bool HasDiscount => Discount > 0;
    
    // Цвет фона по ТЗ
    public string RowBackground => Stock == 0 ? "LightGray" : (Discount > 25 ? "#23E1EF" : "Transparent");
}

```

### 2. `LoginWindow.axaml.cs`

```csharp
using Avalonia.Controls;
using Avalonia.Interactivity;
using System.Linq;

namespace ChitayApp.Views;

public partial class LoginWindow : Window
{
    public LoginWindow() { InitializeComponent(); }

    private void Login_Click(object sender, RoutedEventArgs e)
    {
        var user = DbHelper.Context.Users.FirstOrDefault(u => u.Login == LoginBox.Text && u.Password == PasswordBox.Text);
        if (user != null)
        {
            DbHelper.CurrentUser = user;
            new MainWindow().Show();
            this.Close();
        }
        else ErrorText.Text = "Неверный логин или пароль";
    }

    private void Guest_Click(object sender, RoutedEventArgs e)
    {
        DbHelper.CurrentUser = null; // Гость
        new MainWindow().Show();
        this.Close();
    }
}

```

### 3. `MainWindowViewModel.cs` (Логика списка)

Фильтрация и поиск должны происходить в реальном времени.

```csharp
using System.Collections.ObjectModel;
using System.Linq;
using ReactiveUI;
using ChitayApp.Entities;
using Microsoft.EntityFrameworkCore;

namespace ChitayApp.ViewModels;

public class MainWindowViewModel : ViewModelBase
{
    private string _searchText = "";
    private int _selectedSort = 0;
    private int _selectedDiscount = 0;
    private ObservableCollection<Product> _products;

    public MainWindowViewModel() { LoadData(); }

    [cite_start]// ФИО в правом верхнем углу [cite: 33]
    public string UserFullName => DbHelper.IsGuest ? "Гость" : DbHelper.CurrentUser.FullName;
    
    [cite_start]// Гость не видит поиск и фильтры [cite: 12]
    public bool ShowFilters => !DbHelper.IsGuest;

    public string SearchText
    {
        get => _searchText;
        set { this.RaiseAndSetIfChanged(ref _searchText, value); LoadData(); }
    }
    public int SelectedSort
    {
        get => _selectedSort;
        set { this.RaiseAndSetIfChanged(ref _selectedSort, value); LoadData(); }
    }
    public int SelectedDiscount
    {
        get => _selectedDiscount;
        set { this.RaiseAndSetIfChanged(ref _selectedDiscount, value); LoadData(); }
    }
    public ObservableCollection<Product> Products
    {
        get => _products;
        set => this.RaiseAndSetIfChanged(ref _products, value);
    }

    public void LoadData()
    {
        // Подгружаем товары вместе с категорией
        var query = DbHelper.Context.Products.Include(p => p.Category).AsQueryable();

        // Поиск (если не гость)
        if (!string.IsNullOrWhiteSpace(SearchText))
            query = query.Where(x => x.Title.ToLower().Contains(SearchText.ToLower()) || x.Description.ToLower().Contains(SearchText.ToLower()));

        [cite_start]// Фильтрация [cite: 53, 54]
        if (SelectedDiscount == 1) query = query.Where(x => x.Discount >= 0 && x.Discount < 13);
        if (SelectedDiscount == 2) query = query.Where(x => x.Discount >= 13 && x.Discount < 17);
        if (SelectedDiscount == 3) query = query.Where(x => x.Discount >= 17);

        var list = query.ToList();

        [cite_start]// Сортировка [cite: 52]
        if (SelectedSort == 1) list = list.OrderBy(x => x.Price).ToList();
        if (SelectedSort == 2) list = list.OrderByDescending(x => x.Price).ToList();

        Products = new ObservableCollection<Product>(list);
    }
}

```
